#include <Arduino.h>
#define pinLed1 19 //led claro
#define pinLed2 18 //led escuro

void setup() {
  Serial.begin(9600);
  pinMode(pinLed1, OUTPUT);
  pinMode(pinLed2, OUTPUT);

}




void loop() {
  int sensorLDR = analogRead(4);
  Serial.println(sensorLDR);
  delay(2000);
  if(sensorLDR>500){
    digitalWrite(pinLed1, HIGH);//Ambiente está claro
    digitalWrite(pinLed2, LOW);
  }
  else{
   digitalWrite(pinLed1, LOW);//ambiente está escuro
   digitalWrite(pinLed2, HIGH);
  }


}
