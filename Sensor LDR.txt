#include <Arduino.h>
#define pinLed 18 

void setup() {
  Serial.begin(9600);
  pinMode(pinLed, OUTPUT);
  

}


void loop() {
  int sensorLDR = analogRead(4);
  Serial.println(sensorLDR);
  delay(2000);
  if(sensorLDR>300)
  digitalWrite(pinLed, HIGH);
  
  else
   digitalWrite(pinLed, LOW);
  


}
